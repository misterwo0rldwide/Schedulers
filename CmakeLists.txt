cmake_minimum_required(VERSION 3.10)
project(sched_demo)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra")

# Include directories
include_directories(
    ${PROJECT_SOURCE_DIR}/src
	${PROJECT_SOURCE_DIR}/src/http
    ${PROJECT_SOURCE_DIR}/src/schedulers
	${PROJECT_SOURCE_DIR}/src/data_structures/list
    ${PROJECT_SOURCE_DIR}/src/data_structures/rb_tree
)

# Source files
set(SOURCES
    src/main.cpp
	src/http/server.cpp
    src/schedulers/CFS.cpp
    src/schedulers/RoundRobin.cpp
	src/data_structures/list/CircList.cpp
    src/data_structures/rb_tree/RBTree.cpp
)

add_executable(sched_demo ${SOURCES})
